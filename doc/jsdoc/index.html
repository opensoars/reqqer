reqqer (wip)

Requires made easy.

Install

npm install reqqer

Use

// Old way
var reqs = require('reqqer')({
  https: 'https',
  http: 'http'
});

var https = reqs.https,
    http = reqs.http;
// New way 
const {https, http} = require('reqqer')({https, http});
